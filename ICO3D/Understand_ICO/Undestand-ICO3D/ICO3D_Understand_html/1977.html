1     !------------------------------------------------------------------------------------------------------------
2     !
3     ! Subroutine to calculate the equivalent von Mises stress for plane stress and 3D cases
4     !
5     ! Input: stress - stress array
6     !        sdim - number of stess components
7     ! Output: SVM - Equivalent von Mises stress
8     !
9     !------------------------------------------------------------------------------------------------------------
10     subroutine vonMisesStress(stress,sdim,SVM)
11         implicit none
12         
13         integer(4),intent(IN)::sdim
14         real(8),dimension(sdim,1)::stress
15         real(8),intent(OUT)::SVM
16         real(8)::aux
17         
18         SVM=0.0d0
19         
20         if(sdim==3)then
21             SVM=sqrt(stress(1,1)**2.0d0+stress(2,1)**2.0d0-stress(1,1)*stress(2,1)+3.0d0*stress(3,1)**2.0d0)
22         elseif(sdim==6)then
23             SVM=(stress(1,1)-stress(2,1))**2+(stress(2,1)-stress(3,1))**2+(stress(1,1)-stress(3,1))**2
24             SVM=SVM+6.0d0*(stress(4,1)**2+stress(5,1)**2+stress(6,1)**2)
25             SVM=SVM/2.0d0
26             SVM=sqrt(SVM)
27         end if
28     
29         continue
30     
31     end subroutine
