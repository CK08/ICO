2     ! Subroutine to calculate the inverse of a matrix
3     !
4     ! Input: A(n,n) - Matrix to be inverted
5     !        n      - matrix dimension
6     !        B(n)   - Array of zeros 
7     !        ierr   - error indicator
8     ! Output: A(n,n) - Inverse of the matrix used in the input
9     !--------------------------------------------------------------
10     
11     SUBROUTINE GAUSSJ(A,N,B,IERR)
12     	IMPLICIT REAL(8) (A-H,O-Z)
13     	INTEGER::N
14     	REAL(8),DIMENSION(N,N)::A
15     	REAL(8),DIMENSION(N)::B
16     	INTEGER,PARAMETER::NMAX=50
17     	INTEGER,DIMENSION(N)::INDXC,INDXR,IPIV
18     	IERR=0
19     !	IF(N.GT.NMAX)THEN
20     !	IERR=2
21     !	RETURN
22     !	ENDIF
23     	DO 11 J=1,N
24     	IPIV(J)=0
25     	11 CONTINUE
26     	DO 22 I=1,N
27     	BIG=0.
28     	DO 13 J=1,N
29     	IF(IPIV(J).NE.1)THEN
30     	DO 12 K=1,N
31     	IF(IPIV(K).EQ.0) THEN
32     	IF(ABS(A(J,K)).GE.BIG)THEN
33     	BIG=ABS(A(J,K))
34     	IROW=J
35     	ICOL=K
36     	ENDIF
37     	ELSE IF (IPIV(K).GT.1) THEN
38     	IERR=1
39     	RETURN
40     	ENDIF
41     	12 CONTINUE
42     	ENDIF
43     	13 CONTINUE
44     	IPIV(ICOL)=IPIV(ICOL)+1
45     	IF(IROW.NE.ICOL) THEN
46     	DO 14 L=1,N
47     	DUM=A(IROW,L)
48     	A(IROW,L)=A(ICOL,L)
49     	A(ICOL,L)=DUM
50     	14 CONTINUE
51     	DUM=B(IROW)
52     	B(IROW)=B(ICOL)
53     	B(ICOL)=DUM
54     	ENDIF
55     	INDXR(I)=IROW
56     	INDXC(I)=ICOL
57     	IF(A(ICOL,ICOL).EQ.0.)THEN
58     	IERR=1
59     	ENDIF
60     	PIVINV=1./A(ICOL,ICOL)
61     	A(ICOL,ICOL)=1.
62     	DO 16 L=1,N
63     	A(ICOL,L)=A(ICOL,L)*PIVINV
64     	16 CONTINUE
65     	B(ICOL)=B(ICOL)*PIVINV
66     	DO 21 LL=1,N
67     	IF(LL.NE.ICOL)THEN
68     	DUM=A(LL,ICOL)
69     	A(LL,ICOL)=0.
70     	DO 18 L=1,N
71     	A(LL,L)=A(LL,L)-A(ICOL,L)*DUM
72     	18 CONTINUE
73     	B(LL)=B(LL)-B(ICOL)*DUM
74     	ENDIF
75     	21 CONTINUE
76     	22 CONTINUE
77     	DO 24 L=N,1,-1
78     	IF(INDXR(L).NE.INDXC(L))THEN
79     	DO 23 K=1,N
80     	DUM=A(K,INDXR(L))
81     	A(K,INDXR(L))=A(K,INDXC(L))
82     	A(K,INDXC(L))=DUM
83     	23 CONTINUE
84     	ENDIF
85     	24 CONTINUE
86     END SUBROUTINE GAUSSJ

109     subroutine cross(a,b,axb)
110      
111         implicit none
112         integer,parameter :: wp=8
113         real(wp),dimension(3) :: axb
114         real(wp),dimension(3),intent(in) :: a
115         real(wp),dimension(3),intent(in) :: b 
116      
117         axb(1) = a(2)*b(3) - a(3)*b(2)
118         axb(2) = a(3)*b(1) - a(1)*b(3)
119         axb(3) = a(1)*b(2) - a(2)*b(1)
