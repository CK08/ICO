3     subroutine ElHex27EAS_PW(nel,Ke,el_ddisp,Finte,MatA,MatC,MatV)
4         
5         use Mod_Variables
6         implicit none
7         
8         integer(4)::nel
9         integer(4)::cpi
10         
11         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1),intent(IN)::el_ddisp
12         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::Ke
13         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1),intent(OUT)::Finte
14         
15         real(8),dimension(nalpha,nalpha),intent(OUT)::MatV
16         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::MatA
17         real(8),dimension((p+1)*(q+1)*(w+1)*nds,nalpha),intent(OUT)::MatC
18         
19         integer(4),parameter::npi_xi= 3, npi_eta = 3, npi_zeta = 3
20         
21         integer(4)::i,j,jj,k,k2,k3,k4,k5,k6,k7,k8,ni,nj,nk,nel_nza,k1,error,count,count2
22         real(8)::xi,eta,zeta
23         real(8),dimension(npi_xi)::e,we
24         real(8),dimension(npi_eta)::n,wn
25         real(8),dimension(npi_zeta)::c,wc
26         
27         real(8),dimension((p+1)*(q+1)*(w+1))::R
28         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdx
29         real(8),dimension(6,(p+1)*(q+1)*(w+1)*nds)::Bglob,Bloc
30         real(8),dimension(ntens,1)::tempstr,deform
31         
32         real(8)::detj,detjnew,gwt
33         
34         real(8),dimension(6,6)::matD,MatT
35         real(8),dimension(6,nalpha)::Malpha,Balpha
36         real(8),dimension(nalpha,1)::aux1
37         real(8),dimension(nalpha)::temp1
38         
39         integer(4)::ilixo
40         real(8)::detj0
41         real(8),dimension((p+1)*(q+1)*(w+1))::R0
42         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdx0
43         real(8),dimension(3,3)::jac0,jac0inv
44         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1)::updtdisp
45         real(8),dimension(3)::temp3
46         real(8)::dBdxi,dBdeta,dBdzeta
47         real(8)::a1,a2,a3
48         
49         cpi = 0
50         
51         we = 1.0d0
52         wn = 1.0d0
53         wc = 1.0d0
54         call gauleg(npi_xi, e, we)
55         call gauleg(npi_eta, n, wn)
56         call gauleg(npi_zeta, c, wc)
57         
58         Ke = 0.0d0
59         MatA = 0.0d0
60         MatC = 0.0d0
61         MatV = 0.0d0
62         
63         !Patch centre -------------------
64         updtdisp = 0.0d0
65         call ShapeFunc2(nel,0.0d0,0.0d0,0.0d0,R,dRdx,detj0,jac0,updtdisp)
66     
67         temp3 = 0.0d0
68         ilixo = 0
69         jac0inv = jac0
70         call gaussj (jac0inv,nds,temp3,ilixo)
71         call TransformationMat3D(jac0inv,MatT)
72         
73         
74         !Gauss points cycle -------------------------------------------------------------------------------
75         cpi = 0
76         Ke = 0.0d0
77         Finte = 0.0d0
78         do i=1,npi_xi
79             do j=1,npi_eta
80                 do k=1,npi_zeta
81                 
82                     cpi = cpi + 1
83                     MatD(:,:) = TmatD(nel,cpi,:,:)
84                     
85                     xi = e(i)
86                     eta = n(j)
87                     zeta = c(k)
88                     
89                     call ShapeFunc(nel,xi,eta,zeta,R,dRdx,detj)
90                     
91                     !Weight factor
92                     gwt = we(i)*wn(j)*wc(k)*detj
93                     
94                     !Strain-displacement matrix -------------
95                     Bglob = 0.0d0
96                     do k1=1,(p+1)*(q+1)*(w+1)
97                         Bglob(1,k1*3-2) = dRdx(k1,1)
98                         Bglob(2,k1*3-1) = dRdx(k1,2)
99                         Bglob(3,k1*3  ) = dRdx(k1,3)
100                         Bglob(4,k1*3-2) = dRdx(k1,2)
101                         Bglob(4,k1*3-1) = dRdx(k1,1)
102                         Bglob(5,k1*3-2) = dRdx(k1,3)
103                         Bglob(5,k1*3  ) = dRdx(k1,1)
104                         Bglob(6,k1*3-1) = dRdx(k1,3)
105                         Bglob(6,k1*3  ) = dRdx(k1,2)
106                     end do
107                     
108                     Malpha = 0.0d0
109                     
110                     Malpha(1,1) = -xi*(1.0-eta*eta)*(1.0-zeta*zeta);
111                     Malpha(2,2) = -eta*(1.0-xi*xi)*(1.0-zeta*zeta);
112                     Malpha(3,3) = -zeta*(1.0-xi*xi)*(1.0-eta*eta);
113                     
114                     Malpha(1,4) = 2.0d0*xi*eta*(1.0-zeta*zeta);
115                     Malpha(2,4) = 2.0d0*xi*eta*(1.0-zeta*zeta);
116                     Malpha(3,4) = 2.0d0*xi*eta*(1.0-zeta*zeta);
117                     
118                     Malpha(1,5) = 2.0d0*xi*zeta*(1.0-eta*eta);
119                     Malpha(2,5) = 2.0d0*xi*zeta*(1.0-eta*eta);
120                     Malpha(3,5) = 2.0d0*xi*zeta*(1.0-eta*eta);
121                     
122                     Malpha(1,6) = 2.0d0*eta*zeta*(1.0-xi*xi);
123                     Malpha(2,6) = 2.0d0*eta*zeta*(1.0-xi*xi);
124                     Malpha(3,6) = 2.0d0*eta*zeta*(1.0-xi*xi);
125                 
126                     Malpha(1,7) = 4.0*xi*eta*zeta;
127                     Malpha(2,8) = 4.0*xi*eta*zeta;
128                     Malpha(3,9) = 4.0*xi*eta*zeta;
129                
130                     Malpha(1,10) = -(1.0-eta*eta)*(1.0-zeta*zeta);
131                     Malpha(2,11) = -(1.0-xi*xi)*(1.0-zeta*zeta);
132                     Malpha(3,12) = -(1.0-xi*xi)*(1.0-eta*eta);
133     
134                     
135                     
136                     
137     !                Malpha(4,1) = -xi*(1.0-eta*eta)*(1.0-zeta*zeta);
138     !                Malpha(4,2) = -eta*(1.0-xi*xi)*(1.0-zeta*zeta);
139     !                Malpha(5,3) = -xi*(1.0-eta*eta)*(1.0-zeta*zeta);
140     !                Malpha(5,4) = -zeta*(1.0-xi*xi)*(1.0-eta*eta);
141     !                Malpha(6,5) = -eta*(1.0-xi*xi)*(1.0-zeta*zeta);
142     !                Malpha(6,6) = -zeta*(1.0-xi*xi)*(1.0-eta*eta);
143     !                
144     !                Malpha(4,7) = 2.0d0*xi*zeta*(1.0-eta*eta);
145     !                Malpha(4,8) = 2.0d0*eta*zeta*(1.0-xi*xi);
146     !                Malpha(5,9) = 2.0d0*xi*eta*(1.0-zeta*zeta);
147     !                Malpha(5,10) = 2.0d0*eta*zeta*(1.0-xi*xi);
148     !                Malpha(6,11) = 2.0d0*xi*eta*(1.0-zeta*zeta);
149     !                Malpha(6,12) = 2.0d0*xi*zeta*(1.0-eta*eta);
150                     
151                     Balpha = 0.0d0
152                     Balpha = matmul(MatT,Malpha)*detj0/detj
153                     
154                     
155                     Bloc = Bglob
156                     call MatPlastic3D(iprops,nds,(p+1)*(q+1)*(w+1),props,Bloc,6,el_ddisp,stress(nel,cpi,:),strain(nel,cpi,:),dstrain(nel,cpi,:),dstress(nel,cpi,:),hard(nel,cpi),matD)
157                     
158                     tempstr = 0.0d0
159                     do k1=1,ntens
160                         tempstr(k1,1) = stress(nel,cpi,k1)
161                         deform(k1,1) = dstrain(nel,cpi,k1)
162                     end do
163                     
164                     !Matrix A ------------------------------------------------------------
165                     MatA = MatA + matmul(matmul(transpose(Bglob),MatD),Bglob)*gwt
166                     !---------------------------------------------------------------------
167                     
168                     !Matrix C ------------------------------------------------------------
169                     MatC = MatC + matmul(matmul(transpose(Bglob),MatD),Balpha)*gwt
170                     !---------------------------------------------------------------------
171                     
172                     !Matrix V ------------------------------------------------------------
173                     MatV = MatV + matmul(matmul(transpose(Balpha),MatD),Balpha)*gwt
174                     !---------------------------------------------------------------------
175                     
176                     Ke = Ke + matmul(matmul(transpose(Bglob),matD),Bglob)*gwt
177                     
178                     Finte = Finte + matmul(transpose(Bglob),tempstr)*gwt
179                     
180     !                do k1=1,ntens
181     !                    SEnergy = SEnergy + tempstr(k1,1)*deform(k1,1)*gwt/2.0d0
182     !                end do
183                     
184                     continue
185                     
186                 end do  
187             end do
188         end do
189     
190         continue
191     
192     end subroutine
