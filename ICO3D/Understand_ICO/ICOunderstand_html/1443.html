3     subroutine ElHex27ANS_PW(nel,Ke,el_ddisp,Finte,MatA,MatC,MatV)
4         
5         use Mod_Variables
6         implicit none
7         
8         integer(4)::nel
9         integer(4)::cpi
10         
11         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1),intent(IN)::el_ddisp
12         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::Ke
13         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1),intent(OUT)::Finte
14         
15         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::MatV
16         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::MatA
17         real(8),dimension((p+1)*(q+1)*(w+1)*nds,(p+1)*(q+1)*(w+1)*nds),intent(OUT)::MatC
18         
19         integer(4),parameter::npi_xi= 3, npi_eta = 3, npi_zeta = 3
20         
21         integer(4)::i,j,jj,k,k2,k3,k4,k5,k6,k7,k8,ni,nj,nk,nel_nza,k1,error,count,count2
22         real(8)::xi,eta,zeta
23         real(8),dimension(npi_xi)::e,we
24         real(8),dimension(npi_eta)::n,wn
25         real(8),dimension(npi_zeta)::c,wc
26         
27         real(8),dimension((p+1)*(q+1)*(w+1))::R
28         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdx
29         real(8),dimension(6,(p+1)*(q+1)*(w+1)*nds)::Bglob,Bloc
30         real(8),dimension(ntens,1)::tempstr,deform
31         
32         real(8)::detj,detjnew,gwt
33         
34         real(8),dimension(6,6)::matD,MatT
35         real(8),dimension(6,nalpha)::Malpha,Balpha
36         real(8),dimension(nalpha,1)::aux1
37         real(8),dimension(nds)::temp1
38         
39         integer(4)::ilixo
40         real(8)::detj0
41         real(8),dimension((p+1)*(q+1)*(w+1))::R0
42         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdx0
43         real(8),dimension(3,3)::jac0,jac0inv
44         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1)::updtdisp
45         real(8),dimension(3)::temp3
46         real(8)::dBdxi,dBdeta,dBdzeta
47         real(8)::a1,a2,a3
48         
49         real(8)::detjA,A,B
50         real(8),dimension(nds,nds)::dxdxiA,dxdxi,jacA,jac,jacinv
51         real(8),dimension(6,(p+1)*(q+1)*(w+1)*nds)::BNat,BNat2,BCN2,BCN
52         real(8),dimension((p+1)*(q+1)*(w+1))::RA
53         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdxi,dRdxii
54         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdxA,dRdxiA,dRdxiiA
55         real(8),dimension(16)::shpANS
56         real(8),dimension(16,3)::coordANS
57         real(8),dimension(3,1)::dRdx_teste,temp31
58         real(8),dimension(6,(p+1)*(q+1)*(w+1)*nds)::BNatural,BN2
59         real(8),dimension(nds,nds)::Mat33
60         real(8),dimension(nds*2,nds*2)::TGC,TCL
61         
62         cpi = 0
63         
64         we = 1.0d0
65         wn = 1.0d0
66         wc = 1.0d0
67         call gauleg(npi_xi, e, we)
68         call gauleg(npi_eta, n, wn)
69         call gauleg(npi_zeta, c, wc)
70         
71         updtdisp = 0.0d0
72         
73         Ke = 0.0d0
74         MatA = 0.0d0
75         MatC = 0.0d0
76         MatV = 0.0d0
77         
78         !Gauss points cycle -------------------------------------------------------------------------------
79         cpi = 0
80         Ke = 0.0d0
81         Finte = 0.0d0
82         do i=1,npi_xi
83             do j=1,npi_eta
84                 do k=1,npi_zeta
85                 
86                     cpi = cpi + 1
87                     MatD(:,:) = TmatD(nel,cpi,:,:)
88                     
89                     xi = e(i)
90                     eta = n(j)
91                     zeta = c(k)
92                     
93                     !call ShapeFunc(nel,xi,eta,zeta,R,dRdx,detj)
94                     call ShapeFunc3(nel,xi,eta,zeta,R,dRdx,dRdxi,dRdxii,detj,jac,dxdxi,updtdisp)
95                     
96                     jacinv = 0.0d0
97                     jacinv = jac
98                     call gaussj(jacinv,nds,temp1,ilixo)
99                     
100                     !Global to natural transformation matrix for end configuration
101                     Mat33 = jac
102                     call TransformationMat3D(mat33,TGC)
103                     
104                     Mat33 = jacinv
105                     call TransformationMat3D(mat33,TCL)
106                     
107                     !Weight factor
108                     gwt = we(i)*wn(j)*wc(k)*detj
109                     
110                     !Strain-displacement matrix -------------
111                     Bglob = 0.0d0
112                     do k1=1,(p+1)*(q+1)*(w+1)
113                         Bglob(1,k1*3-2) = dRdx(k1,1)
114                         Bglob(2,k1*3-1) = dRdx(k1,2)
115                         Bglob(3,k1*3  ) = dRdx(k1,3)
116                         !Bglob(4,k1*3-2) = dRdx(k1,2)
117                         !Bglob(4,k1*3-1) = dRdx(k1,1)
118                         Bglob(5,k1*3-2) = dRdx(k1,3)
119                         Bglob(5,k1*3  ) = dRdx(k1,1)
120                         Bglob(6,k1*3-1) = dRdx(k1,3)
121                         Bglob(6,k1*3  ) = dRdx(k1,2)
122                     end do
123                     
124                     BNatural = matmul(TGC,Bglob)
125                     
126                     !ANS Stuff
127                     BCN = BNatural
128                     
129                     !ANS Implementation I
130                     A = sqrt(1.0d0/3.0d0)
131                     B = sqrt(3.0d0/5.0d0)
132                     
133                     !Coordinates
134                     coordANS = 0.0d0
135                     coordANS(1,1)  =     A; coordANS(1,2)  =     B
136                     coordANS(2,1)  =    -A; coordANS(2,2)  =     B
137                     coordANS(3,1)  =     A; coordANS(3,2)  = 0.0d0
138                     coordANS(4,1)  =    -A; coordANS(4,2)  = 0.0d0
139                     coordANS(5,1)  =     A; coordANS(5,2)  =    -B
140                     coordANS(6,1)  =    -A; coordANS(6,2)  =    -B
141                     coordANS(7,1)  =     B; coordANS(7,2)  =     A
142                     coordANS(8,1)  = 0.0d0; coordANS(8,2)  =     A
143                     coordANS(9,1)  =    -B; coordANS(9,2)  =     A
144                     coordANS(10,1) =     B; coordANS(10,2) =    -A
145                     coordANS(11,1) = 0.0d0; coordANS(11,2) =    -A
146                     coordANS(12,1) =    -B; coordANS(12,2) =    -A
147                     coordANS(13,1) =     A; coordANS(13,2) =     A
148                     coordANS(14,1) =    -A; coordANS(14,2) =     A
149                     coordANS(15,1) =     A; coordANS(15,2) =    -A
150                     coordANS(16,1) =    -A; coordANS(16,2) =    -A
151                     !-------------------------------------------------------------------
152                     !-------------------------------------------------------------------
153                     
154                     !Shape Functions
155                     shpANS = 0.0d0
156                     !-------------------------------------------------------------------
157                     shpANS(1)  = eta/(4.0d0*B)*(eta/B+1.0d0)*(1.0d0+xi/A) 
158                     shpANS(2)  = eta/(4.0d0*B)*(eta/B+1.0d0)*(1.0d0-xi/A)
159                     shpANS(3)  = 1.0d0/2.0d0*(1.0d0-eta*eta/(B*B))*(1.0d0+xi/A)
160                     shpANS(4)  = 1.0d0/2.0d0*(1.0d0-eta*eta/(B*B))*(1.0d0-xi/A)
161                     shpANS(5)  = eta/(4.0d0*B)*(eta/B-1.0d0)*(1.0d0+xi/A) 
162                     shpANS(6)  = eta/(4.0d0*B)*(eta/B-1.0d0)*(1.0d0-xi/A)
163                     !-------------------------------------------------------------------
164                     
165                     !-------------------------------------------------------------------
166                     shpANS(7)  = xi/(4.0d0*B)*(xi/B+1.0d0)*(1.0d0+eta/A)
167                     shpANS(8)  = 1.0d0/2.0d0*(1.0d0-xi*xi/(B*B))*(1.0d0+eta/A)
168                     shpANS(9)  = xi/(4.0d0*B)*(xi/B-1.0d0)*(1.0d0+eta/A)
169                     shpANS(10) = xi/(4.0d0*B)*(xi/B+1.0d0)*(1.0d0-eta/A)
170                     shpANS(11) = 1.0d0/2.0d0*(1.0d0-xi*xi/(B*B))*(1.0d0-eta/A)
171                     shpANS(12) = xi/(4.0d0*B)*(xi/B-1.0d0)*(1.0d0-eta/A)
172                     !-------------------------------------------------------------------
173     
174                     !-------------------------------------------------------------------
175                     shpANS(13) = 1.0d0/4.0d0*(1.0d0+xi/A)*(1.0d0+eta/A)
176                     shpANS(14) = 1.0d0/4.0d0*(1.0d0-xi/A)*(1.0d0+eta/A)
177                     shpANS(15) = 1.0d0/4.0d0*(1.0d0+xi/A)*(1.0d0-eta/A)
178                     shpANS(16) = 1.0d0/4.0d0*(1.0d0-xi/A)*(1.0d0-eta/A)
179                     !-------------------------------------------------------------------
180                     
181                     BNat = 0.0d0
182                     
183                     !BNat(1,:) = BCN(1,:)
184                     !BNat(2,:) = BCN(2,:)
185                     !BNat(3,:) = BCN(3,:)
186                     !BNat(4,:) = BCN(4,:)
187                     !BNat(5,:) = BCN(5,:)
188                     !BNat(6,:) = BCN(6,:)
189                     
190                     !xi-eta components
191                     do k1=13,16
192                         Bnat(4,:) = Bnat(4,:) + shpANS(k1)*BCN(4,:)
193                     end do
194                     
195                     Bloc=matmul(TCL,BNat)
196                     
197     
198                     
199                     Bloc = Bglob
200                     call MatPlastic3D(iprops,nds,(p+1)*(q+1)*(w+1),props,Bloc,6,el_ddisp,stress(nel,cpi,:),strain(nel,cpi,:),dstrain(nel,cpi,:),dstress(nel,cpi,:),hard(nel,cpi),matD)
201                     
202                     tempstr = 0.0d0
203                     do k1=1,ntens
204                         tempstr(k1,1) = stress(nel,cpi,k1)
205                         deform(k1,1) = dstrain(nel,cpi,k1)
206                     end do
207                     
208                     !Matrix A ------------------------------------------------------------
209                     MatA = MatA + matmul(matmul(transpose(Bglob),MatD),Bglob)*gwt
210                     !---------------------------------------------------------------------
211                     
212                     !Matrix C ------------------------------------------------------------
213                     MatC = MatC + matmul(matmul(transpose(Bglob),MatD),Bnat)*gwt
214                     !---------------------------------------------------------------------
215                     
216                     !Matrix V ------------------------------------------------------------
217                     MatV = MatV + matmul(matmul(transpose(Bnat),MatD),Bnat)*gwt
218                     !---------------------------------------------------------------------
219                     
220                     Ke = Ke + matmul(matmul(transpose(Bglob),matD),Bglob)*gwt
221                     
222                     Finte = Finte + matmul(transpose(Bglob),tempstr)*gwt
223                     
224     !                do k1=1,ntens
225     !                    SEnergy = SEnergy + tempstr(k1,1)*deform(k1,1)*gwt/2.0d0
226     !                end do
227                     
228                     continue
229                     
230                 end do  
231             end do
232         end do
233     
234         continue
235     
236     end subroutine
