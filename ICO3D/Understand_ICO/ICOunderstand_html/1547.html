4     !-----[--.----+----.----+----.-----------------------------------------]
5     !      Purpose: Compute eigenvalues/vectors for 3x3 symmetric matrix
6     !
7     !      Inputs:
8     !         v(3,3) - matrix with initial values (only upper half used)
9     !
10     !      Outputs:
11     !         v(3,3) - matrix of eigenvectors (by column)
12     !         d(3)   - eigenvalues associated with columns of v
13     !         rot    - number of rotations to diagonalize
14     !-----[--.----+----.----+----.-----------------------------------------]
15     !     Storage done as follows:
16     !
17     !       | v(1,1) v(1,2) v(1,3) |     |  d(1)  a(1)  a(3)  |
18     !       | v(2,1) v(2,2) v(2,3) |  =  |  a(1)  d(2)  a(2)  |
19     !       | v(3,1) v(3,2) v(3,3) |     |  a(3)  a(2)  d(3)  |
20     !
21     !        Transformations performed on d(i) and a(i) and v(i,j) become
22     !        eigenvectors.  Thus, original array is destroyed.
23     !-----[--.----+----.----+----.-----------------------------------------]     
24           
25           subroutine eig3(v,d,rot)
26     
27           implicit  none
28     
29           integer   rot, its, i,j,k
30           real*8    g,h, aij, sm,thresh, t, c,s,tau
31           real*8    v(3,3), d(3), a(3), b(3), z(3)
32     
33     !     Move array into one-d arrays
34     
35           a(1) = v(1,2)
36           a(2) = v(2,3)
37           a(3) = v(1,3)
38     
39           do i = 1,3
40             d(i) = v(i,i)
41             b(i) = d(i)
42             z(i) = 0.0d0
43             do j = 1,3
44               v(i,j) = 0.0d0
45             end do ! j
46             v(i,i) = 1.d0
47           end do ! i
48     
49           rot = 0
50     
51           do its = 1,50
52     
53     !       Set convergence test and threshold
54     
55             sm = abs(a(1)) + abs(a(2)) + abs(a(3))
56             if (sm.eq.0.0d0) return
57     
58             if(its.lt.4) then
59               thresh = 0.011d0*sm
60             else
61               thresh = 0.0d0
62             end if
63     
64     !       Perform sweeps for rotations
65     
66             do i = 1,3
67               j = mod(i,3) + 1
68               k = mod(j,3) + 1
69     
70               aij  = a(i)
71               g    = 100.d0*abs(aij)
72               if(abs(d(i))+g.ne.abs(d(i)) .or. abs(d(j))+g.ne.abs(d(j))) then
73     
74                 if(abs(aij).gt.thresh) then
75                   a(i) = 0.0d0
76                   h    = d(j) - d(i)
77                   if(abs(h)+g.eq.abs(h)) then
78                     t = aij/h
79                   else
80                     t = sign(2.d0,h/aij)/(abs(h/aij)+sqrt(4.d0+(h/aij)**2))
81                   endif
82     !
83     !ccccccccccccccccccccccccccccccccccccccccc
84                   if(t.eq.1.d10) write(*,*)t
85     !ccccccccccccccccccccccccccccccccccccccccc	      
86     !
87     !             Set rotation parameters
88     
89                   c    = 1.d0/sqrt(1.d0+t*t)
90                   s    = t*c
91                   tau  = s/(1.d0+c)
92     
93     !             Rotate diagonal terms
94     
95                   h    = t*aij
96                   z(i) = z(i) - h
97                   z(j) = z(j) + h
98                   d(i) = d(i) - h
99                   d(j) = d(j) + h
100     
101     !             Rotate off-diagonal terms
102     
103                   h    = a(j)
104                   g    = a(k)
105                   a(j) = h + s*(g - h*tau)
106                   a(k) = g - s*(h + g*tau)
107     
108     !             Rotate eigenvectors
109     
110                   do k = 1,3
111                     g      = v(k,i)
112                     h      = v(k,j)
113                     v(k,i) = g - s*(h + g*tau)
114                     v(k,j) = h + s*(g - h*tau)
115                   end do ! k
116     
117                   rot = rot + 1
118     
119                 end if
120               else
121                 a(i) = 0.0d0
122               end if
123             end do ! i
124     
125     !       Update diagonal terms
126     
127             do i = 1,3
128               b(i) = b(i) + z(i)
129               d(i) = b(i)
130               z(i) = 0.0d0
131             end do ! i
132     
133           end do ! its
134     
135           end
136           
137           
138           !*** DETERMINA OS VALORES PRPRIOS E
139     !*** VECTORES PRPRIOS DE UMA
140     !*** MATRIZ SIMTRICA 2x2 OU 3x3
141     !*** NDI{I}:DIMENSO DO PROBLEMA
142     !*** A(NDI,NDI){O}:MATRIZ SIMTRICA
143     !*** VLP(NDI){O}:VALORES PRPRIOS DECRESCENTES
144     !*** VCP(NDI,NDI){O}:VECTORES PRPRIOS POR COLUNA
145     !*** IER{O}:INDICADOR DE ERRO
146     !*** [0]:NO OCORREU QUALQUER ERRO
147     !*** [1]:DIMENSO DO PROBLEMA  ILEGAL
148     !*** [2]:ERRO NA DECOMPOSIO ESPECTRAL
149     !
150     	SUBROUTINE VECP23(NDI,A,VLP,VCP,IER)
151     !	USE NUMERICAL_LIBRARIES
152     	IMPLICIT REAL(8) (A-H,O-Z)
153     	REAL(8),DIMENSION(NDI,NDI)::A,VCP,B
154     	REAL(8),DIMENSION(NDI)::VLP
155     	IER=0
156     !*** VERIFICA AS DIMENSES
157     	IF(NDI.NE.2.AND.NDI.NE.3)THEN
158     	IER=1
159     	RETURN
160     	ENDIF
161     !*** MTODO DE JACOBI
162     	!CALL DEVCSF (NDI, A, NDI, VLP,VCP,NDI)
163     	CALL JACOB(A,VLP,VCP,NDI,IER) 
164     	IF(IER.NE.0)THEN
165     	IER=2
166     	ENDIF
167     	END SUBROUTINE VECP23
168     !
169     !*** DECOMPOSIO ESPECTRAL DE UMA
170     !*** MATRIZ SIMTRICA
171     !*** A(N,N)...........MATRIZ A DECOMPOR 
172     !*** D(N).............VALORES PRPRIOS  
173     !*** V(N,N)...........VECTORES PRPRIOS 
174     !*** IERR.............INDICADOR DE ERRO 
175     !
176     	SUBROUTINE JACOB(A,D,V,N,IERR) 
177     	IMPLICIT REAL(8) (A-H,O-Z)
178     	REAL(8),DIMENSION(N,N)::A,V
179     	REAL(8),DIMENSION(N)::D
180     	REAL(8),DIMENSION(N)::B,Z
181     	REAL(8),PARAMETER::RZE=0.0D00,RUM=1.0D00,RCE=100.0D00
182     	INTEGER,PARAMETER::MITER=900
183     	IERR=0
184     	DO IP=1,N
185     	DO IQ=1,N
186     	V(IP,IQ)=RZE
187     	ENDDO
188     	V(IP,IP)=RUM
189     	ENDDO
190     	DO IP=1,N
191     	B(IP)=A(IP,IP)
192     	D(IP)=B(IP)
193     	Z(IP)=RZE
194     	ENDDO
195     	DO I=1,MITER
196     	SM=RZE
197     	DO IP=1,N-1
198     	DO IQ=IP+1,N
199     	SM=SM+ABS(A(IP,IQ))
200     	ENDDO
201     	ENDDO
202     	IF(SM.EQ.RZE)RETURN
203     	IF(I.LT.4)THEN
204     	TRESH=0.2D00*SM/N**2
205     	ELSE
206     	TRESH=RZE
207     	ENDIF
208     	DO IP=1,N-1
209     	DO IQ=IP+1,N
210     	G=RCE*ABS(A(IP,IQ))
211     	IF((I.GT.4).AND.(ABS(D(IP))+ G.EQ.ABS(D(IP))).AND.(ABS(D(IQ))+G.EQ.ABS(D(IQ))))THEN
212     	A(IP,IQ)=RZE
213     	ELSE IF(ABS(A(IP,IQ)).GT.TRESH)THEN
214     	H=D(IQ)-D(IP)
215     	IF(ABS(H)+G.EQ.ABS(H))THEN
216     	T=A(IP,IQ)/H
217     	ELSE
218     	THETA=0.5D00*H/A(IP,IQ)
219     	T=1.0D00/(ABS(THETA)+SQRT(RUM+THETA**2))
220     	IF(THETA.LT.RZE)T=-T
221     	ENDIF
222     	C=RUM/SQRT(RUM+T**2)
223     	S=T*C
224     	TAU=S/(RUM+C)
225     	H=T*A(IP,IQ)
226     	Z(IP)=Z(IP)-H
227     	Z(IQ)=Z(IQ)+H
228     	D(IP)=D(IP)-H
229     	D(IQ)=D(IQ)+H
230     	A(IP,IQ)=RZE
231     	DO J=1,IP-1
232     	G=A(J,IP)
233     	H=A(J,IQ)
234     	A(J,IP)=G-S*(H+G*TAU)
235     	A(J,IQ)=H+S*(G-H*TAU)
236     	ENDDO
237     	DO J=IP+1,IQ-1
238     	G=A(IP,J)
239     	H=A(J,IQ)
240     	A(IP,J)=G-S*(H+G*TAU)
241     	A(J,IQ)=H+S*(G-H*TAU)
242     	ENDDO
243     	DO J=IQ+1,N
244     	G=A(IP,J)
245     	H=A(IQ,J)
246     	A(IP,J)=G-S*(H+G*TAU)
247     	A(IQ,J)=H+S*(G-H*TAU)
248     	ENDDO
249     	DO J=1,N
250     	G=V(J,IP)
251     	H=V(J,IQ)
252     	V(J,IP)=G-S*(H+G*TAU)
253     	V(J,IQ)=H+S*(G-H*TAU)
254     	ENDDO
255     	ENDIF
256     	ENDDO
257     	ENDDO
258     	DO IP=1,N
259     	B(IP)=B(IP)+Z(IP)
260     	D(IP)=B(IP)
261     	Z(IP)=RZE
262     	ENDDO
263     	ENDDO
264     	IERR=1
265     	RETURN
266     	END SUBROUTINE JACOB
