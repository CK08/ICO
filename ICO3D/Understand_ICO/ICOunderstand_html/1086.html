1     !--------------------------------------------------------------------------
2     !   Calculate basis functions of a B-Spline
3     !   Based on the algorithm A2.2 from "The NURBS book"
4     !    
5     !   Input:
6     !       s - knot span
7     !       u - data point
8     !       p - B-Spline degree
9     !       u_knots - knot vector
10     !   Output:
11     !       BF(p+1,1) - Non-zero basis functions
12     !--------------------------------------------------------------------------
13     
14     subroutine BasisFuns(nb,s,u,p,u_knots,BF)
15         
16         implicit none
17         
18         integer(4),intent(IN)::s,p,nb
19         real(8),intent(IN)::u
20         real(8),dimension(nb),intent(IN)::u_knots
21         real(8),dimension(p+1,1),intent(OUT)::BF
22         
23         integer(4)::i,j,r,k1
24         real(8)::saved,temp
25         real(8),dimension(p+1)::N,left,right
26         
27     
28         N = 0.0d0
29         BF = 0.0d0
30         left = 0.0d0
31         right = 0.0d0
32     
33         N(1) = 1.0;
34     
35         i = s;
36     
37         do j=1,p
38             left(j+1) = u - u_knots(i+1-j)
39             right(j+1) = u_knots(i+j) - u
40             saved = 0.0
41             
42             do r=0,j-1
43                 temp = N(r+1)/(right(r+2)+left(j-r+1))
44                 N(r+1) = saved+right(r+2)*temp
45                 saved = left(j-r+1)*temp
46             end do
47             
48             N(j+1) = saved;
49             continue
50             
51         end do
52     
53         do k1=1,p+1
54             BF(k1,1) = N(k1);
55         end do
56     
57         return
58     
59         
60         
61         
62     end subroutine BasisFuns
