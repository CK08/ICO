1     !------------------------------------------------------------------------------------
2     !
3     ! Subroutine to create the mesh file for GiD, based on the control polygon
4     !
5     !------------------------------------------------------------------------------------
6     
7     subroutine GiDEqMesh(coord)
8         
9         use Mod_Variables
10         implicit none
11         
12         integer(4)::i,j,l,k,k1,uind,nodpelem,count,nbx,nby,counter
13         integer(4)::nelx,nely,el1,el2,el3,el4
14         real(8)::ucheck,vcheck,den
15         integer(4)::uspan,vspan,idx
16         real(8),dimension(p+1,1)::BFx
17         real(8),dimension(q+1,1)::BFy
18         real(8),dimension(1,1,nds+1)::temp
19         character(16)::my_type
20         
21         real(8),dimension(q+1)::Ny
22         real(8),dimension(q+1)::dNydxi
23         
24         !Nodes coordinates (size overestimated)
25         real(8),dimension(ncpx*ncpy,nds)::coord
26         
27         open(unit=2, file='IGAEq.flavia.msh')
28         
29         if(elcode == 'Quad4E' .or. elcode == 'Quad9E' .or. elcode == 'Quad4S' .or. elcode == 'Quad9S') then
30             my_type = "Quadrilateral"
31             nodpelem = 4
32         endif
33         
34         write(2,FMT=111)nds, my_type, nodpelem
35         
36         !Heading ----------------------------------------------------------------
37         111 format('MESH "problem" dimension ', I1,' ElemType ',A13, ' Nnode ', I2)
38         
39         !Coordinates ---------------------------
40         counter=0
41         write(2,*)'coordinates'
42         
43         nbx = ncpx+p+1
44         nby = ncpy+q+1
45         
46         nelx = ncpx - p
47         nely = ncpy - q
48         
49         coord = 0.0d0
50     !    do i=1,ncpy
51     !        do j=1,nds
52     !            coord(i,j) = b_net(1,i,j)
53     !        enddo
54     !    end do
55         
56         !--------------------------------------------------------------------------------
57         !u = 0.0
58         ucheck = 0.0d0
59         call FindSpan(nbx,p,ucheck,U_Knot,uspan)  
60         BFx = 0.0d0
61         call BasisFuns(nbx,uspan,ucheck,p,u_knot,BFx) 
62         
63         vcheck = 0.0d0   
64         call FindSpan(nby,q,vcheck,V_Knot,vspan)
65         BFy = 0.0d0
66         call BasisFuns(nby,vspan,vcheck,q,v_knot,BFy)
67         
68         counter = counter + 1
69         den = 0.0;
70         j = 1
71         do l=0,q
72             temp = 0.0d0
73             do k=0,p
74                 do k1=1,nds+1
75                     temp(1,1,k1)  = temp(1,1,k1)  + BFx(k+1,1)*Bw(uspan-p+k, vspan-q+l, k1)
76                 enddo
77                 continue
78             enddo
79             
80             den = den + temp(1,1,nds+1)*BFy(l+1,1);
81             do k1=1,nds
82                 coord(counter,k1) = coord(counter,k1) + temp(1,1,k1)*BFy(l+1,1);
83             enddo
84         enddo
85         
86         do j=1,ncpy+q
87             if(v_knot(j) .ne. v_knot(j+1))then
88                 vcheck = v_knot(j+1)
89                 
90                 call FindSpan(nby,q,vcheck,V_Knot,vspan)
91         
92                 BFy = 0.0d0
93                 call BasisFuns(nby,vspan,vcheck,q,v_knot,BFy)
94     
95                 counter = counter + 1
96                 den = 0.0;
97                 do l=0,q
98                     temp = 0.0d0 !zeros(1,1,3);
99                     do k=0,p
100                         do k1=1,nds+1
101                             temp(1,1,k1)  = temp(1,1,k1)  + BFx(k+1,1)*Bw(uspan-p+k, vspan-q+l, k1);
102                         enddo
103                         continue
104                     enddo
105                     
106                     den = den + temp(1,1,nds+1)*BFy(l+1,1);
107                     do k1=1,nds
108                         coord(counter,k1) = coord(counter,k1) + temp(1,1,k1)*BFy(l+1,1);
109                     end do
110                 enddo
111                 
112                 do k1=1,nds
113                     coord(counter,k1) = coord(counter,k1)/den;
114                 end do
115                 
116                 continue
117                 
118             endif
119         end do
120         
121         !--------------------------------------------------------------------------------
122         !Search u_knot
123         !counter = ncpy
124         do i=1,ncpx+p
125             if(u_knot(i) .ne. u_knot(i+1))then
126                 ucheck = u_knot(i+1)
127                 
128                 call FindSpan(nbx,p,ucheck,U_Knot,uspan)
129                 
130                 BFx = 0.0d0
131                 call BasisFuns(nbx,uspan,ucheck,p,u_knot,BFx)
132                 
133                 
134                 vcheck = 0.0d0
135                         
136                 call FindSpan(nby,q,vcheck,V_Knot,vspan)
137         
138                 BFy = 0.0d0
139                 call BasisFuns(nby,vspan,vcheck,q,v_knot,BFy)
140                 
141                 counter = counter + 1
142                 den = 0.0;
143                 j = 1
144                 do l=0,q
145                     temp = 0.0d0
146                     do k=0,p
147                         do k1=1,nds+1
148                             temp(1,1,k1)  = temp(1,1,k1)  + BFx(k+1,1)*Bw(uspan-p+k, vspan-q+l, k1)
149                         enddo
150                         continue
151                     enddo
152                     
153                     den = den + temp(1,1,nds+1)*BFy(l+1,1);
154                     do k1=1,nds
155                         coord(counter,k1) = coord(counter,k1) + temp(1,1,k1)*BFy(l+1,1);
156                     enddo
157                 enddo
158                 
159                 do k1=1,nds
160                     coord(counter,k1) = coord(counter,k1)/den;
161                 enddo
162                 
163                 do j=1,ncpy+q
164                     if(v_knot(j) .ne. v_knot(j+1))then
165                         vcheck = v_knot(j+1)
166                         
167                         call FindSpan(nby,q,vcheck,V_Knot,vspan)
168                 
169                         BFy = 0.0d0
170                         call BasisFuns(nby,vspan,vcheck,q,v_knot,BFy)
171     
172                         counter = counter + 1
173                         den = 0.0;
174                         do l=0,q
175                             temp = 0.0d0 !zeros(1,1,3);
176                             do k=0,p
177                                 do k1=1,nds+1
178                                     temp(1,1,k1)  = temp(1,1,k1)  + BFx(k+1,1)*Bw(uspan-p+k, vspan-q+l, k1);
179                                 enddo
180                                 continue
181                             enddo
182                             
183                             den = den + temp(1,1,nds+1)*BFy(l+1,1);
184                             do k1=1,nds
185                                 coord(counter,k1) = coord(counter,k1) + temp(1,1,k1)*BFy(l+1,1);
186                             end do
187                         enddo
188                         
189                         do k1=1,nds
190                             coord(counter,k1) = coord(counter,k1)/den;
191                         end do
192                         
193                         continue
194                         
195                     endif
196                 end do
197                 
198                 continue
199             end if
200     
201         end do
202         !--------------------------------------------------------------------------------
203         
204         do j=1,ncpx*ncpy
205             write(2,FMT=112)j,coord(j,:)
206             112 format(I4,5(E,1x))
207         end do
208     
209         write(2,*)'end coordinates'
210         
211         !Elements ------------------------------
212         count = 0
213         write(2,*)'Elements'
214         
215         do i=1,nelx
216             do j = 1, nely
217             count = count + 1
218             idx = i + nely*(i-1) + (j-1)
219             el1 = idx
220             el2 = idx + 1
221             el3 = idx + nely + 2
222             el4 = idx + nely + 1
223             
224             write(2,FMT=113)count,el1,el2,el3,el4
225             113 format(21(I4,1x))
226             end do
227         end do
228         
229         write(2,*)'end elements'
230         
231         close(2,status='keep')
232         
233         continue
234         
235         
236     end subroutine
