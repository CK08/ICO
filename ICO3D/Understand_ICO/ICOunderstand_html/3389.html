1     !----------------------------------------------------------------------------------------------
2     !
3     ! Axis Update subroutine
4     !
5     ! Input: raux - Axis Matrix
6     !        matR - Rotation Matrix
7     !
8     ! Output: rconv - Updated axis
9     !
10     !----------------------------------------------------------------------------------------------
11     subroutine axisupdate(raux,matr,rconv)
12             implicit none
13             
14             real(8),dimension(3,3),intent(IN)::raux,matr
15             real(8),dimension(3,3),intent(OUT)::rconv
16             integer(4)::j
17             real(8),dimension(3,1)::aux31,auxx,auxy,auxz
18             
19             do j=1,3
20                 aux31(j,1)=raux(j,1)
21             end do
22             
23             auxx=matmul(matR,aux31)
24             
25             do j=1,3
26                 aux31(j,1)=raux(j,2)
27             end do
28             
29             auxy=matmul(matR,aux31)
30             
31             call cross(auxx,auxy,auxz)
32                     
33             do j=1,3
34                 rconv(j,1)=auxx(j,1)
35                 rconv(j,2)=auxy(j,1)
36                 rconv(j,3)=auxz(j,1)
37             end do
38     
39     
40         end subroutine axisupdate
