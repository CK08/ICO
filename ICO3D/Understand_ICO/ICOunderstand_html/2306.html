1     !------------------------------------------------------------------------------------------------------------
2     !
3     ! General quadrilateral element for plain strain problems with a 4x4 gaussian quadrature
4     !
5     ! Input: idx1, idx2 - NURBS coordinates (element indexes in the index space) (currently not being used)
6     !        nel - Element number
7     !
8     ! Output: Ke - Elemental stiffness matrix
9     !------------------------------------------------------------------------------------------------------------
10     
11     
12     subroutine ElQuad16E(nel,Ke,el_ddisp,Finte)
13         
14         use Mod_Variables
15         implicit none
16         
17         integer(4)::nel
18         integer(4)::cpi
19         
20         real(8),dimension((p+1)*(q+1)*nds,1),intent(IN)::el_ddisp
21         real(8),dimension((p+1)*(q+1)*nds,(p+1)*(q+1)*nds),intent(OUT)::Ke
22         real(8),dimension((p+1)*(q+1)*nds,1),intent(OUT)::Finte
23         
24         
25         integer(4),parameter::npi_xi= 4, npi_eta = 4
26         
27         integer(4)::i,j,ni,nj,nk,nel_nza,k1
28         real(8)::xi,eta
29         real(8),dimension(npi_xi)::e,we
30         real(8),dimension(npi_eta)::n,wn
31         
32         real(8),dimension((p+1)*(q+1))::R
33         real(8),dimension((p+1)*(q+1),nds)::dRdx
34         real(8)::detJ,gwt
35         
36         real(8),dimension(3,(p+1)*(q+1)*nds)::Bglob,Bloc
37         
38         real(8),dimension(3,3)::matD
39         real(8),dimension(ntens,1)::tempstr
40         
41         cpi = 0
42     
43         we=1.0d0
44         wn=1.0d0
45         call gauleg(npi_xi, e, we)
46         call gauleg(npi_eta, n, wn)
47         
48         !Gauss points cycle
49         Ke = 0.0d0
50         Finte = 0.0d0
51         do i=1,npi_xi
52             do j=1,npi_eta
53                 
54                 cpi = cpi + 1
55                 MatD(:,:) = TmatD(nel,cpi,:,:)
56                 
57                 xi = e(i)
58                 eta = n(j)
59                 
60                 call ShapeFunc(nel,xi,eta,R,dRdx,detj)
61                 
62                 !Weight factor
63                 gwt = we(i)*wn(j)*detj
64                 
65                 Bglob = 0.0d0
66                 do k1=1,(p+1)*(q+1)
67                     Bglob(1,k1*2-1) = dRdx(k1,1)
68                     Bglob(2,k1*2  ) = dRdx(k1,2)
69                     Bglob(3,k1*2-1) = dRdx(k1,2)
70                     Bglob(3,k1*2  ) = dRdx(k1,1)
71                 end do
72                 
73                 Bloc = Bglob
74                 
75                 !call MaterialPStrain(ndi,nshr,ntens,iprops,nodedof,nnodes,props,B,Blines,ddisp,stress,strain,dstrain,dstress,hard,matD,matf0,matf,dtempo)
76                 call MaterialPStrain(iprops,nds,(p+1)*(q+1),props,Bloc,3,el_ddisp,stress(nel,cpi,:),strain(nel,cpi,:),dstrain(nel,cpi,:),dstress(nel,cpi,:),hard(nel,cpi),matD)
77                 
78                 tempstr = 0.0d0
79                 do k1=1,ntens
80                     tempstr(k1,1) = stress(nel,cpi,k1)
81                 end do
82                 
83                 Ke = Ke + matmul(matmul(transpose(Bglob),matD),Bglob)*gwt
84                 
85                 Finte = Finte + matmul(transpose(Bloc),tempstr)*gwt
86                 
87                 continue
88               
89             end do
90         end do
91     
92         continue
93     
94     end subroutine ElQuad16E
