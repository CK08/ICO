1     subroutine ShapeFuncBubble(nel,xii,etai,zetai,R,dRdx,detj,jac,udisp)
2         
3         use Mod_Variables
4         
5         implicit none
6         
7         !Element number
8         integer(4)::nel,idx1,idx2,idx3
9         real(8),intent(IN)::xii,etai,zetai
10         !Array of bivariate NURBS basis functions
11         real(8),dimension((p+1)*(q+1)*(w+1)),intent(OUT)::R
12         !Array of bivariate NURBS basis functions derivatives wrt physical coordinates
13         real(8),dimension((p+1)*(q+1)*(w+1),nds),intent(OUT)::dRdx
14         !Jacobian Determinant
15         real(8),intent(OUT)::detJ
16         real(8),dimension(nds,nds),intent(OUT)::jac
17         real(8),dimension((p+1)*(q+1)*(w+1)*nds,1),intent(IN)::udisp
18         
19         real(8)::detj1,detj2
20         
21         !NURBS coordinates
22         integer(4)::ni,nj,nk,i,j,k,aa,bb,cc,loc_num,errorflag
23         real(8)::sumxi,sumeta,sumzeta,sumtot    !,nk
24         real(8),dimension(nds)::temp1
25         
26         !Parametric coordinates
27         real(8)::xi,eta,zeta
28         
29         !Arrays of univariate B-Spline basis functions
30         real(8),dimension(p+1)::N
31         real(8),dimension(q+1)::M
32         real(8),dimension(w+1)::L
33         
34         !Basis functions derivatives wrt parametric coordinates
35         real(8),dimension(p+1)::dNdxi
36         real(8),dimension(q+1)::dMdeta
37         real(8),dimension(w+1)::dLdzeta
38         
39         !Derivative of the physical coordinates wrt parametric coordinates and its inverse
40         real(8),dimension(nds,nds)::dxidx
41         
42         !Derivatives of parametric coordinates wrt parent element coordinates
43         real(8),dimension(nds,nds)::dxi_dtildexi
44     
45         integer(4)::count
46         real(8)::tmp,tol
47         
48         real(8),dimension((p+1)*(q+1)*(w+1),nds)::dRdxi,dRdxii
49         real(8),dimension(nds,nds)::dxdxi
50         
51         tol = 1.0d-3
52         
53         !Initializations
54         R = 0.0d0
55         dRdx = 0.0d0
56         dRdxi = 0.0d0
57         detJ = 0.0d0
58         
59         ni = 0
60         nj = 0
61         nk = 0
62         
63         xi   = 0.0d0
64         eta  = 0.0d0
65         zeta = 0.0d0
66         
67         N = 0.0d0
68         M = 0.0d0
69         L = 0.0d0
70         
71         dNdxi = 0.0d0
72         dMdeta = 0.0d0
73         dLdzeta = 0.0d0
74         
75         dxdxi = 0.0d0
76         dxidx = 0.0d0
77         
78         dxi_dtildexi = 0.0d0
79         
80         i = 0
81         j = 0
82         k = 0
83         
84         aa = 0
85         bb = 0
86         cc = 0
87         
88         loc_num = 0
89         
90         jac     = 0.0d0
91         sumxi   = 0.0d0
92         sumeta  = 0.0d0
93         sumzeta = 0.0d0
94         sumtot  = 0.0d0
95         
96         !NURBS coordinates for local node 1
97         ni = INN(IEN(nel,1),1)
98         nj = INN(IEN(nel,1),2)
99         nk = INN(IEN(nel,1),3)
100         
101         !Calculate location of the integration point in the parametric space (xi,eta) based on the coordinates of the parent element (xii,etai)
102         xi   = 0.5d0 !((u_knot(ni+1) - u_knot(ni))*xii   + (u_knot(ni+1) + u_knot(ni)))/2.0d0
103         eta  = 0.5d0 !((v_knot(nj+1) - v_knot(nj))*etai  + (v_knot(nj+1) + v_knot(nj)))/2.0d0
104         zeta = 0.5d0 !((w_knot(nk+1) - w_knot(nk))*zetai + (w_knot(nk+1) + w_knot(nk)))/2.0d0
105         
106         !Determinant of the jacobian parametric-parent
107         detj1 = (U_knot(ni+1)-U_knot(ni))*(V_knot(nj+1)-V_knot(nj))*(W_knot(nk+1)-W_knot(nk))/8.0d0
108         
109         !Calculate Basis functions and derivatives
110         call BSplineBasisAndDeriv(ncpx,p,xi,u_knot,N,dNdxi)
111         call BSplineBasisAndDeriv(ncpy,q,eta,v_knot,M,dMdeta)
112         call BSplineBasisAndDeriv(ncpz,w,zeta,w_knot,L,dLdzeta)
113         
114         
115         !Build numerators and denominators
116         
117         do k=0,w
118             do j=0,q
119                 do i=0,p
120                     loc_num = loc_num + 1     !Local basis function number
121                     R(loc_num) = N(p+1-i)*M(q+1-j)*L(w+1-k) !*weight(IEN(nel,loc_num)) !B_net(ni-i,nj-j,nk-k,nds+1)
122                     sumtot = sumtot + R(loc_num)
123                     
124                     dRdxi(loc_num,1) = dNdxi(p+1-i)*M(q+1-j)*L(w+1-k) !*weight(IEN(nel,loc_num)) !*B_net(ni-i,nj-j,nk-k,nds+1)
125                     sumxi = sumxi + dRdxi(loc_num,1)
126                     
127                     dRdxi(loc_num,2) = N(p+1-i)*dMdeta(q+1-j)*L(w+1-k) !*weight(IEN(nel,loc_num)) !*B_net(ni-i,nj-j,nk-k,nds+1)
128                     sumeta = sumeta + dRdxi(loc_num,2)
129                     
130                     dRdxi(loc_num,3) = N(p+1-i)*M(q+1-j)*dLdzeta(w+1-k) !*weight(IEN(nel,loc_num)) !*B_net(ni-i,nj-j,nk-k,nds+1)
131                     sumzeta = sumzeta + dRdxi(loc_num,3)
132                 end do
133             end do
134         end do
135         
136         
137         do i=1,(p+1)*(q+1)*(w+1)
138             dRdxi(i,1) = dRdxi(i,1)/sumtot - (R(i)*sumxi  )/(sumtot**2)
139             dRdxi(i,2) = dRdxi(i,2)/sumtot - (R(i)*sumeta )/(sumtot**2)
140             dRdxi(i,3) = dRdxi(i,3)/sumtot - (R(i)*sumzeta)/(sumtot**2) 
141             R(i) = R(i)/sumtot
142         end do
143         
144       
145         !Gradient of mapping from parameter space to physical space
146         loc_num = 0
147         count = (p+1)*(q+1)*(w+1)+1
148         do k=0,w
149             do j=0,q
150                 do i=0,p
151                     
152                     loc_num = loc_num + 1
153                     count = count - 1
154                     
155                     dxdxi(1,1) = dxdxi(1,1) + dRdxi(loc_num,1)*(Points(IEN(nel,loc_num),1) + udisp(loc_num*3-2,1))
156                     dxdxi(2,1) = dxdxi(2,1) + dRdxi(loc_num,1)*(Points(IEN(nel,loc_num),2) + udisp(loc_num*3-1,1))
157                     dxdxi(3,1) = dxdxi(3,1) + dRdxi(loc_num,1)*(Points(IEN(nel,loc_num),3) + udisp(loc_num*3  ,1))
158                     
159                     dxdxi(1,2) = dxdxi(1,2) + dRdxi(loc_num,2)*(Points(IEN(nel,loc_num),1) + udisp(loc_num*3-2,1))
160                     dxdxi(2,2) = dxdxi(2,2) + dRdxi(loc_num,2)*(Points(IEN(nel,loc_num),2) + udisp(loc_num*3-1,1))
161                     dxdxi(3,2) = dxdxi(3,2) + dRdxi(loc_num,2)*(Points(IEN(nel,loc_num),3) + udisp(loc_num*3  ,1))
162                     
163                     dxdxi(1,3) = dxdxi(1,3) + dRdxi(loc_num,3)*(Points(IEN(nel,loc_num),1) + udisp(loc_num*3-2,1))
164                     dxdxi(2,3) = dxdxi(2,3) + dRdxi(loc_num,3)*(Points(IEN(nel,loc_num),2) + udisp(loc_num*3-1,1))
165                     dxdxi(3,3) = dxdxi(3,3) + dRdxi(loc_num,3)*(Points(IEN(nel,loc_num),3) + udisp(loc_num*3  ,1))
166                     
167                 end do
168             end do
169         end do
170         
171     
172         call fdeterm(dxdxi,detj2)
173         
174         !Inverse of the gradient
175         temp1 = 0.0d0
176         dxidx = dxdxi
177         call gaussj(dxidx,nds,temp1,errorflag)
178         
179         dRdx = matmul(dRdxi,dxidx)
180         
181         detj = detj1*detj2 
182         
183         detj = abs(detj)
184         
185         !Gradient mapping from parent element to parameter space
186         dxi_dtildexi(1,1) = (U_knot(ni+1)-U_knot(ni))/2.0d0
187         dxi_dtildexi(2,2) = (V_knot(nj+1)-V_knot(nj))/2.0d0
188         dxi_dtildexi(3,3) = (W_knot(nk+1)-W_knot(nk))/2.0d0
189         
190         dRdxii = matmul(dRdxi,dxi_dtildexi)
191         
192         Jac = 0.0d0
193         do aa=1,nds
194             do bb=1,nds
195                 do cc=1,nds
196                     Jac(aa,bb) = Jac(aa,bb) + dxdxi(aa,cc)*dxi_dtildexi(cc,bb)
197                 end do
198             end do
199         end do
200         
201         !Gradient mapping from parent element to physical space 
202     !    jac = 0.0d0
203     !    jac = matmul(dxdxi,dxi_dtildexi)
204         
205         !jac = dxdxi
206         
207         continue
208         
209     end subroutine
