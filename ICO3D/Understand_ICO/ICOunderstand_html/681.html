5           subroutine SYMMLQ( n, b, r1, r2, v, w, x, y,
6          $                   Aprod, Msolve, checkA, goodb, precon, shift,
7          $                   nout , itnlim, rtol,
8          $                   istop, itn, Anorm, Acond, rnorm, ynorm )
9     
10           external           Aprod, Msolve
11           integer            n, nout, itnlim, istop, itn
12           logical            checkA, goodb, precon
13           double precision   shift, rtol, Anorm, Acond, rnorm, ynorm,
14          $                   b(n), r1(n), r2(n), v(n), w(n), x(n), y(n)
15     *     ------------------------------------------------------------------
16     *
17     *     SYMMLQ  is designed to solve the system of linear equations
18     *
19     *                Ax = b
20     *
21     *     where A is an n by n symmetric matrix and b is a given vector.
22     *     The matrix A is not required to be positive definite.
23     *     (If A is known to be definite, the method of conjugate gradients
24     *     might be preferred, since it will require about the same number of
25     *     iterations as SYMMLQ but slightly less work per iteration.)
26     *
27     *
28     *     The matrix A is intended to be large and sparse.  It is accessed
29     *     by means of a subroutine call of the form
30     *
31     *                call Aprod ( n, x, y )
32     *
33     *     which must return the product y = Ax for any given vector x.
34     *
35     *
36     *     More generally, SYMMLQ is designed to solve the system
37     *
38     *                (A - shift*I) x = b
39     *
40     *     where  shift  is a specified scalar value.  If  shift  and  b
41     *     are suitably chosen, the computed vector x may approximate an
42     *     (unnormalized) eigenvector of A, as in the methods of
43     *     inverse iteration and/or Rayleigh-quotient iteration.
44     *     Again, the matrix (A - shift*I) need not be positive definite.
45     *     The work per iteration is very slightly less if  shift = 0.
46     *
47     *
48     *     A further option is that of preconditioning, which may reduce
49     *     the number of iterations required.  If M = C C' is a positive
50     *     definite matrix that is known to approximate  (A - shift*I)
51     *     in some sense, and if systems of the form  My = x  can be
52     *     solved efficiently, the parameters precon and Msolve may be
53     *     used (see below).  When  precon = .true., SYMMLQ will
54     *     implicitly solve the system of equations
55     *
56     *             P (A - shift*I) P' xbar  =  P b,
57     *
58     *     i.e.                  Abar xbar  =  bbar
59     *     where                         P  =  C**(-1),
60     *                                Abar  =  P (A - shift*I) P',
61     *                                bbar  =  P b,
62     *
63     *     and return the solution       x  =  P' xbar.
64     *     The associated residual is rbar  =  bbar - Abar xbar
65     *                                      =  P (b - (A - shift*I)x)
66     *                                      =  P r.
67     *
68     *     In the discussion below, eps refers to the machine precision.
69     *     eps is computed by SYMMLQ.  A typical value is eps = 2.22e-16
70     *     for IBM mainframes and IEEE double-precision arithmetic.
71     *
72     *     Parameters
73     *     ----------
74     *
75     *     n       input      The dimension of the matrix A.
76     *
77     *     b(n)    input      The rhs vector b.
78     *
79     *     r1(n)   workspace
80     *     r2(n)   workspace
81     *     v(n)    workspace
82     *     w(n)    workspace
83     *
84     *     x(n)    output     Returns the computed solution  x.
85     *
86     *     y(n)    workspace
87     *
88     *     Aprod   external   A subroutine defining the matrix A.
89     *                        For a given vector x, the statement
90     *
91     *                              call Aprod ( n, x, y )
92     *
93     *                        must return the product y = Ax
94     *                        without altering the vector x.
95     *
96     *     Msolve  external   An optional subroutine defining a
97     *                        preconditioning matrix M, which should
98     *                        approximate (A - shift*I) in some sense.
99     *                        M must be positive definite.
100     *                        For a given vector x, the statement
101     *
102     *                              call Msolve( n, x, y )
103     *
104     *                        must solve the linear system My = x
105     *                        without altering the vector x.
106     *
107     *                        In general, M should be chosen so that Abar has
108     *                        clustered eigenvalues.  For example,
109     *                        if A is positive definite, Abar would ideally
110     *                        be close to a multiple of I.
111     *                        If A or A - shift*I is indefinite, Abar might
112     *                        be close to a multiple of diag( I  -I ).
113     *
114     *                        NOTE.  The program calling SYMMLQ must declare
115     *                        Aprod and Msolve to be external.
116     *
117     *     checkA  input      If checkA = .true., an extra call of Aprod will
118     *                        be used to check if A is symmetric.  Also,
119     *                        if precon = .true., an extra call of Msolve
120     *                        will be used to check if M is symmetric.
121     *
122     *     goodb   input      Usually, goodb should be .false.
123     *                        If x is expected to contain a large multiple of
124     *                        b (as in Rayleigh-quotient iteration),
125     *                        better precision may result if goodb = .true.
126     *                        See Lewis (1977) below.
127     *                        When goodb = .true., an extra call to Msolve
128     *                        is required.
129     *
130     *     precon  input      If precon = .true., preconditioning will
131     *                        be invoked.  Otherwise, subroutine Msolve
132     *                        will not be referenced; in this case the
133     *                        actual parameter corresponding to Msolve may
134     *                        be the same as that corresponding to Aprod.
135     *
136     *     shift   input      Should be zero if the system Ax = b is to be
137     *                        solved.  Otherwise, it could be an
138     *                        approximation to an eigenvalue of A, such as
139     *                        the Rayleigh quotient b'Ab / (b'b)
140     *                        corresponding to the vector b.
141     *                        If b is sufficiently like an eigenvector
142     *                        corresponding to an eigenvalue near shift,
143     *                        then the computed x may have very large
144     *                        components.  When normalized, x may be
145     *                        closer to an eigenvector than b.
146     *
147     *     nout    input      A file number.
148     *                        If nout .gt. 0, a summary of the iterations
149     *                        will be printed on unit nout.
150     *
151     *     itnlim  input      An upper limit on the number of iterations.
152     *
153     *     rtol    input      A user-specified tolerance.  SYMMLQ terminates
154     *                        if it appears that norm(rbar) is smaller than
155     *                              rtol * norm(Abar) * norm(xbar),
156     *                        where rbar is the transformed residual vector,
157     *                              rbar = bbar - Abar xbar.
158     *
159     *                        If shift = 0 and precon = .false., SYMMLQ
160     *                        terminates if norm(b - A*x) is smaller than
161     *                              rtol * norm(A) * norm(x).
162     *
163     *     istop   output     An integer giving the reason for termination...
164     *
165     *              -1        beta2 = 0 in the Lanczos iteration; i.e. the
166     *                        second Lanczos vector is zero.  This means the
167     *                        rhs is very special.
168     *                        If there is no preconditioner, b is an
169     *                        eigenvector of A.
170     *                        Otherwise (if precon is true), let My = b.
171     *                        If shift is zero, y is a solution of the
172     *                        generalized eigenvalue problem Ay = lambda My,
173     *                        with lambda = alpha1 from the Lanczos vectors.
174     *
175     *                        In general, (A - shift*I)x = b
176     *                        has the solution         x = (1/alpha1) y
177     *                        where My = b.
178     *                        
179     *               0        b = 0, so the exact solution is x = 0.
180     *                        No iterations were performed.
181     *
182     *               1        Norm(rbar) appears to be less than
183     *                        the value  rtol * norm(Abar) * norm(xbar).
184     *                        The solution in  x  should be acceptable.
185     *
186     *               2        Norm(rbar) appears to be less than
187     *                        the value  eps * norm(Abar) * norm(xbar).
188     *                        This means that the residual is as small as
189     *                        seems reasonable on this machine.
190     *
191     *               3        Norm(Abar) * norm(xbar) exceeds norm(b)/eps,
192     *                        which should indicate that x has essentially
193     *                        converged to an eigenvector of A
194     *                        corresponding to the eigenvalue shift.
195     *
196     *               4        Acond (see below) has exceeded 0.1/eps, so
197     *                        the matrix Abar must be very ill-conditioned.
198     *                        x may not contain an acceptable solution.
199     *
200     *               5        The iteration limit was reached before any of
201     *                        the previous criteria were satisfied.
202     *
203     *               6        The matrix defined by Aprod does not appear
204     *                        to be symmetric.
205     *                        For certain vectors y = Av and r = Ay, the
206     *                        products y'y and r'v differ significantly.
207     *
208     *               7        The matrix defined by Msolve does not appear
209     *                        to be symmetric.
210     *                        For vectors satisfying My = v and Mr = y, the
211     *                        products y'y and r'v differ significantly.
212     *
213     *               8        An inner product of the form  x' M**(-1) x
214     *                        was not positive, so the preconditioning matrix
215     *                        M does not appear to be positive definite.
216     *
217     *                        If istop .ge. 5, the final x may not be an
218     *                        acceptable solution.
219     *
220     *     itn     output     The number of iterations performed.
221     *
222     *     Anorm   output     An estimate of the norm of the matrix operator
223     *                        Abar = P (A - shift*I) P',   where P = C**(-1).
224     *
225     *     Acond   output     An estimate of the condition of Abar above.
226     *                        This will usually be a substantial
227     *                        under-estimate of the true condition.
228     *
229     *     rnorm   output     An estimate of the norm of the final
230     *                        transformed residual vector,
231     *                           P (b  -  (A - shift*I) x).
232     *
233     *     ynorm   output     An estimate of the norm of xbar.
234     *                        This is sqrt( x'Mx ).  If precon is false,
235     *                        ynorm is an estimate of norm(x).
236     *
237     *
238     *
239     *     To change precision
240     *     -------------------
241     *
242     *     Alter the words
243     *            double precision,
244     *            daxpy, dcopy, ddot, dnrm2
245     *     to their single or double equivalents.

250     *     the following references:
251     *
252     *     C.C. Paige and M.A. Saunders,  Solution of Sparse Indefinite
253     *          Systems of Linear Equations,
254     *          SIAM J. Numer. Anal. 12, 4, September 1975, pp. 617-629.
255     *
256     *     J.G. Lewis,  Algorithms for Sparse Matrix Eigenvalue Problems,
257     *          Report STAN-CS-77-595, Computer Science Department,
258     *          Stanford University, Stanford, California, March 1977.
259     *
260     *     Applications of SYMMLQ and the theory of preconditioning
261     *     are described in the following references:
262     *
263     *     D.B. Szyld and O.B. Widlund,  Applications of Conjugate Gradient
264     *          Type Methods to Eigenvalue Calculations,
265     *          in R. Vichnevetsky and R.S. Steplman (editors),
266     *          Advances in Computer Methods for Partial Differential
267     *          Equations -- III, IMACS, 1979, 167-173.
268     *
269     *     D.B. Szyld,  A Two-level Iterative Method for Large Sparse
270     *          Generalized Eigenvalue Calculations,
271     *          Ph. D. dissertation, Department of Mathematics,
272     *          New York University, New York, October 1983.
273     *
274     *     P.E. Gill, W. Murray, D.B. Ponceleon and M.A. Saunders,
275     *          Preconditioners for indefinite systems arising in
276     *          optimization, SIMAX 13, 1, 292--311, January 1992.
277     *          (SIAM J. on Matrix Analysis and Applications)
278     *     ------------------------------------------------------------------
279     *
280     *
281     *     SYMMLQ development:
282     *            1972: First version.
283     *            1975: John Lewis recommended modifications to help with
284     *                  inverse iteration:
285     *                  1. Reorthogonalize v1 and v2.
286     *                  2. Regard the solution as x = x1  +  bstep * b,
287     *                     with x1 and bstep accumulated separately
288     *                     and bstep * b added at the end.
289     *                     (In inverse iteration, b might be close to the
290     *                     required x already, so x1 may be a lot smaller
291     *                     than the multiple of b.)
292     *            1978: Daniel Szyld and Olof Widlund implemented the first
293     *                  form of preconditioning.
294     *                  This required both a solve and a multiply with M.
295     *            1979: Implemented present method for preconditioning.
296     *                  This requires only a solve with M.
297     *            1984: Sven Hammarling noted corrections to tnorm and x1lq.
298     *                  SYMMLQ added to NAG Fortran Library.
299     *     15 Sep 1985: Final F66 version.  SYMMLQ sent to "misc" in netlib.
300     *     16 Feb 1989: First F77 version.
301     *
302     *     22 Feb 1989: Hans Mittelmann observed beta2 = 0 (hence failure)
303     *                  if Abar = const*I.  istop = -1 added for this case.
304     *
305     *     01 Mar 1989: Hans Mittelmann observed premature termination on
306     *                  ( 1  1  1 )     (   )                   ( 1  1    )
307     *                  ( 1  1    ) x = ( 1 ),  for which  T3 = ( 1  1  1 ).
308     *                  ( 1     1 )     (   )                   (    1  1 )
309     *                  T2 is exactly singular, so estimating cond(A) from
310     *                  the diagonals of Lbar is unsafe.  We now use
311     *                  L       or  Lbar         depending on whether
312     *                  lqnorm  or  cgnorm       is least.
313     *
314     *     03 Mar 1989: eps computed internally instead of coming in as a
315     *                  parameter.
316     *     07 Jun 1989: ncheck added as a parameter to say if A and M
317     *                  should be checked for symmetry.
318     *                  Later changed to checkA (see below).
319     *     20 Nov 1990: goodb added as a parameter to make Lewis's changes
320     *                  an option.  Usually b is NOT much like x.  Setting
321     *                  goodb = .false. saves a call to Msolve at the end.
322     *     20 Nov 1990: Residual not computed exactly at end, to save time
323     *                  when only one or two iterations are required
324     *                  (e.g. if the preconditioner is very good).
325     *                  Beware, if precon is true, rnorm estimates the
326     *                  residual of the preconditioned system, not Ax = b.
327     *     04 Sep 1991: Parameter list changed and reordered.
328     *                  integer ncheck is now logical checkA.
329     *     22 Jul 1992: Example from Lothar Reichel and Daniela Calvetti
330     *                  showed that beta2 = 0 (istop = -1) means that
331     *                  b is an eigenvector when M = I.
332     *                  More complicated if there is a preconditioner;
333     *                  not clear yet how to describe it.
334     *     20 Oct 1999: Bug.  alfa1 = 0 caused Anorm = 0, divide by zero.
335     *                  Need to estimate Anorm from column of Tk.

338     *     Department of EESOR                  mike@sol-michael.stanford.edu

343     *
344     *     Subroutines and functions
345     *
346     *     USER       Aprod, Msolve
347     *     BLAS       daxpy, dcopy, ddot , dnrm2
348     *
349     *
350     *     Intrinsics and local variables
351     
352           intrinsic          abs, max, min, mod, sqrt
353           double precision   ddot, dnrm2
354           double precision   alfa, b1, beta, beta1, bstep, cs,
355          $                   cgnorm, dbar, delta, denom, diag,
356          $                   eps, epsa, epsln, epsr, epsx,
357          $                   gamma, gbar, gmax, gmin, gpert,
358          $                   lqnorm, oldb, qrnorm, rhs1, rhs2,
359          $                   s, sn, snprod, t, tnorm,
360          $                   x1cg, x1lq, ynorm2, zbar, z
361           integer            i
362     
363           double precision   zero         ,  one         ,  two
364           parameter        ( zero = 0.0d+0,  one = 1.0d+0,  two = 2.0d+0 )
365     
366           character*16       enter, exit
367           character*52       msg(-1:8)
368     
369           data               enter /' Enter SYMMLQ.  '/,
370          $                   exit  /' Exit  SYMMLQ.  '/
371     
372           data               msg
373          $ / 'beta2 = 0.  If M = I, b and x are eigenvectors of A',
374          $   'beta1 = 0.  The exact solution is  x = 0',
375          $   'Requested accuracy achieved, as determined by rtol',
376          $   'Reasonable accuracy achieved, given eps',
377          $   'x has converged to an eigenvector',
378          $   'Acond has exceeded 0.1/eps',
379          $   'The iteration limit was reached',
380          $   'Aprod  does not define a symmetric matrix',
381          $   'Msolve does not define a symmetric matrix',
382          $   'Msolve does not define a pos-def preconditioner' /
383     *     ------------------------------------------------------------------
384     
385     
386     *     Compute eps, the machine precision.  The call to daxpy is
387     *     intended to fool compilers that use extra-length registers.
388     
389           eps    = one / 16.0d+0
390     
391        10 eps    = eps / two
392           x(1)   = eps
393           y(1)   = one
394           call daxpy ( 1, one, x, 1, y, 1 )
395           if (y(1) .gt. one) go to 10
396     
397           eps    = eps * two
398     
399     *     Print heading and initialize.
400     
401           if (nout .gt. 0) then
402              write(nout, 1000) enter, n, checkA, goodb, precon,
403          $                     itnlim, rtol, shift
404           end if
405           istop  = 0
406           itn    = 0
407           Anorm  = zero
408           Acond  = zero
409           rnorm  = zero
410           ynorm  = zero
411     
412           do 50 i = 1, n
413              x(i) = zero
414        50 continue
415     
416     *     Set up y for the first Lanczos vector v1.
417     *     y is really beta1 * P * v1  where  P = C**(-1).
418     *     y and beta1 will be zero if b = 0.
419     
420           call dcopy ( n, b, 1, y , 1 )
421           call dcopy ( n, b, 1, r1, 1 )
422           if ( precon ) call Msolve( n, r1, y )
423           if ( goodb  ) then
424              b1  = y(1)
425           else
426              b1  = zero
427           end if
428           beta1  = ddot  ( n, r1, 1, y, 1 )
429     
430     *     See if Msolve is symmetric.
431     
432           if (checkA  .and.  precon) then
433              call Msolve( n, y, r2 )
434              s      = ddot  ( n, y, 1, y, 1 )
435              t      = ddot  ( n,r1, 1,r2, 1 )
436              z      = abs( s - t )
437              epsa   = (s + eps) * eps**0.33333
438              if (z .gt. epsa) then
439                 istop = 7
440                 go to 900
441              end if
442           end if
443     
444     *     Test for an indefinite preconditioner.
445     
446           if (beta1 .lt. zero) then
447              istop = 8
448              go to 900
449           end if
450     
451     *     If b = 0 exactly, stop with x = 0.
452     
453           if (beta1 .eq. zero) then
454              go to 900
455           end if
456     
457     *     Here and later, v is really P * (the Lanczos v).
458     
459           beta1  = sqrt( beta1 )
460           s      = one / beta1
461           do 100 i = 1, n
462              v(i)  = s * y(i)
463       100 continue
464     
465     *     See if Aprod  is symmetric.
466     
467           call Aprod ( n, v, y )
468           if (checkA) then
469              call Aprod ( n, y, r2 )
470              s      = ddot  ( n, y, 1, y, 1 )
471              t      = ddot  ( n, v, 1,r2, 1 )
472              z      = abs( s - t )
473              epsa   = (s + eps) * eps**0.33333
474              if (z .gt. epsa) then
475                 istop = 6
476                 go to 900
477              end if
478           end if
479     
480     *     Set up y for the second Lanczos vector.
481     *     Again, y is beta * P * v2  where  P = C**(-1).
482     *     y and beta will be zero or very small if b is an eigenvector.
483     
484           call daxpy ( n, (- shift), v, 1, y, 1 )
485           alfa   = ddot  ( n, v, 1, y, 1 )
486           call daxpy ( n, (- alfa / beta1), r1, 1, y, 1 )
487     
488     *     Make sure  r2  will be orthogonal to the first  v.
489     
490           z      = ddot  ( n, v, 1, y, 1 )
491           s      = ddot  ( n, v, 1, v, 1 )
492           call daxpy ( n, (- z / s), v, 1, y, 1 )
493     
494           call dcopy ( n, y, 1, r2, 1 )
495           if ( precon ) call Msolve( n, r2, y )
496           oldb   = beta1
497           beta   = ddot  ( n, r2, 1, y, 1 )
498     
499           if (beta .lt. zero) then
500              istop = 8
501              go to 900
502           end if
503     
504     *     Cause termination (later) if beta is essentially zero.
505     
506           beta   = sqrt( beta )
507           if (beta .le. eps) then
508              istop = -1
509           end if
510     
511     *     See if the local reorthogonalization achieved anything.
512     
513           denom  = sqrt( s ) * dnrm2( n, r2, 1 )  +  eps
514           s      = z / denom
515           t      = ddot  ( n, v, 1, r2, 1 ) / denom
516           if (nout .gt. 0  .and.  goodb) then
517              write(nout, 1100) beta1, alfa, s, t
518           end if
519     
520     *     Initialize other quantities.
521     
522           cgnorm = beta1
523           gbar   = alfa
524           dbar   = beta
525           rhs1   = beta1
526           rhs2   = zero
527           bstep  = zero
528           snprod = one
529           tnorm  = alfa**2 + beta**2
530           ynorm2 = zero
531           gmax   = abs( alfa ) + eps
532           gmin   = gmax
533     
534           if ( goodb ) then
535              do 200 i = 1, n
536                 w(i)  = zero
537       200    continue
538           else
539              call dcopy ( n, v, 1, w, 1 )
540           end if
541     
542     *     ------------------------------------------------------------------
543     *     Main iteration loop.
544     *     ------------------------------------------------------------------
545     
546     *     Estimate various norms and test for convergence.
547     
548       300 Anorm  = sqrt( tnorm  )
549           ynorm  = sqrt( ynorm2 )
550           epsa   = Anorm * eps
551           epsx   = Anorm * ynorm * eps
552           epsr   = Anorm * ynorm * rtol
553           diag   = gbar
554           if (diag .eq. zero) diag = epsa
555     
556           lqnorm = sqrt( rhs1**2 + rhs2**2 )
557           qrnorm = snprod * beta1
558           cgnorm = qrnorm * beta / abs( diag )
559     
560     *     Estimate  cond(A).
561     *     In this version we look at the diagonals of  L  in the
562     *     factorization of the tridiagonal matrix,  T = L*Q.
563     *     Sometimes, T(k) can be misleadingly ill-conditioned when
564     *     T(k+1) is not, so we must be careful not to overestimate Acond.
565     
566           if (lqnorm .le. cgnorm) then
567              Acond  = gmax / gmin
568           else
569              denom  = min( gmin, abs( diag ) )
570              Acond  = gmax / denom
571           end if
572     
573     *     See if any of the stopping criteria are satisfied.
574     *     In rare cases, istop is already -1 from above (Abar = const * I).
575     
576           if (istop .eq. 0) then
577              if (itn    .ge. itnlim ) istop = 5
578              if (Acond  .ge. 0.1/eps) istop = 4
579              if (epsx   .ge. beta1  ) istop = 3
580              if (cgnorm .le. epsx   ) istop = 2
581              if (cgnorm .le. epsr   ) istop = 1
582           end if
583     *     ==================================================================
584     
585     *     See if it is time to print something.
586     
587           if (nout .le.  0)          go to 600
588           if (n    .le. 40)          go to 400
589           if (itn  .le. 10)          go to 400
590           if (itn  .ge. itnlim - 10) go to 400
591           if (mod(itn,10)  .eq.   0) go to 400
592           if (cgnorm .le. 10.0*epsx) go to 400
593           if (cgnorm .le. 10.0*epsr) go to 400
594           if (Acond  .ge. 0.01/eps ) go to 400
595           if (istop  .ne. 0)         go to 400
596           go to 600
597     
598     *     Print a line for this iteration.
599     
600       400 zbar   = rhs1 / diag
601           z      = (snprod * zbar  +  bstep) / beta1
602           x1lq   = x(1)  +  b1 * bstep / beta1
603           x1cg   = x(1)  +  w(1) * zbar  +  b1 * z
604     
605           if (    itn     .eq. 0) write(nout, 1200)
606           write(nout, 1300) itn, x1cg, cgnorm, bstep/beta1, Anorm, Acond
607           if (mod(itn,10) .eq. 0) write(nout, 1500)
608     *     ==================================================================
609     
610     
611     *     Obtain the current Lanczos vector  v = (1 / beta)*y
612     *     and set up  y  for the next iteration.
613     
614       600 if (istop .ne. 0) go to 800
615           s      = one / beta
616     
617           do 620 i = 1, n
618              v(i)  = s * y(i)
619       620 continue
620     
621           call Aprod ( n, v, y )
622           call daxpy ( n, (- shift), v, 1, y, 1 )
623           call daxpy ( n, (- beta / oldb), r1, 1, y, 1 )
624           alfa   = ddot( n, v, 1, y, 1 )
625           call daxpy ( n, (- alfa / beta), r2, 1, y, 1 )
626           call dcopy ( n, r2, 1, r1, 1 )
627           call dcopy ( n, y, 1, r2, 1 )
628           if ( precon ) call Msolve( n, r2, y )
629           oldb   = beta
630           beta   = ddot  ( n, r2, 1, y, 1 )
631           if (beta .lt. zero) then
632              istop = 6
633              go to 800
634           end if
635           beta   = sqrt( beta )
636           tnorm  = tnorm  +  alfa**2  +  oldb**2  +  beta**2
637     
638     *     Compute the next plane rotation for  Q.
639     
640           gamma  = sqrt( gbar**2 + oldb**2 )
641           cs     = gbar / gamma
642           sn     = oldb / gamma
643           delta  = cs * dbar  +  sn * alfa
644           gbar   = sn * dbar  -  cs * alfa
645           epsln  = sn * beta
646           dbar   =            -  cs * beta
647     
648     *     Update  x.
649     
650           z      = rhs1 / gamma
651           s      = z * cs
652           t      = z * sn
653     
654           do 700 i = 1, n
655              x(i)  = (w(i) * s   +   v(i) * t)  +  x(i)
656              w(i)  =  w(i) * sn  -   v(i) * cs
657       700 continue
658     
659     *     Accumulate the step along the direction  b,
660     *     and go round again.
661     
662           bstep  = snprod * cs * z  +  bstep
663           snprod = snprod * sn
664           gmax   = max( gmax, gamma )
665           gmin   = min( gmin, gamma )
666           ynorm2 = z**2  +  ynorm2
667           rhs1   = rhs2  -  delta * z
668           rhs2   =       -  epsln * z
669           itn    = itn   +  1
670           go to 300
671     
672     *     ------------------------------------------------------------------
673     *     End of main iteration loop.
674     *     ------------------------------------------------------------------
675     
676     *     Move to the CG point if it seems better.
677     *     In this version of SYMMLQ, the convergence tests involve
678     *     only cgnorm, so we're unlikely to stop at an LQ point,
679     *     EXCEPT if the iteration limit interferes.
680     
681       800 if (cgnorm .le. lqnorm) then
682              zbar   = rhs1 / diag
683              bstep  = snprod * zbar  +  bstep
684              ynorm  = sqrt( ynorm2  +  zbar**2 )
685              rnorm  = cgnorm
686              call daxpy ( n, zbar, w, 1, x, 1 )
687           else
688              rnorm  = lqnorm
689           end if
690     
691           if ( goodb ) then
692     
693     *        Add the step along  b.
694     
695              bstep  = bstep / beta1
696              call dcopy ( n, b, 1, y, 1 )
697              if ( precon ) call Msolve( n, b, y )
698              call daxpy ( n, bstep, y, 1, x, 1 )
699           end if
700     
701     *     ==================================================================
702     *     Display final status.
703     *     ==================================================================
704       900 if (nout  .gt. 0) then
705              write(nout, 2000) exit, istop, itn,
706          $                     exit, Anorm, Acond,
707          $                     exit, rnorm, ynorm
708              write(nout, 3000) exit, msg(istop)
709           end if
710     
711           return
712     
713     *     ------------------------------------------------------------------
714      1000 format(// 1p,    a, 5x, 'Solution of symmetric   Ax = b'
715          $       / ' n      =', i7, 5x, 'checkA =', l4, 12x,
716          $          'goodb  =', l4, 7x, 'precon =', l4
717          $       / ' itnlim =', i7, 5x, 'rtol   =', e11.2, 5x,
718          $          'shift  =', e23.14)
719      1100 format(/ 1p, ' beta1  =', e10.2, 3x, 'alpha1 =', e10.2
720          $       / ' (v1,v2) before and after ', e14.2
721          $       / ' local reorthogonalization', e14.2)
722      1200 format(// 5x, 'itn', 7x, 'x1(cg)', 10x,
723          $         'norm(r)', 5x, 'bstep', 7x, 'norm(A)', 3X, 'cond(A)')
724      1300 format(1p, i8, e19.10, e11.2, e14.5, 2e10.2)
725      1500 format(1x)
726      2000 format(/ 1p, a, 6x, 'istop =', i3,   15x, 'itn   =', i8
727          $       /     a, 6x, 'Anorm =', e12.4, 6x, 'Acond =', e12.4
728          $       /     a, 6x, 'rnorm =', e12.4, 6x, 'ynorm =', e12.4)
729      3000 format(      a, 6x, a )
730     *     ------------------------------------------------------------------
731     *     end of SYMMLQ
732           end

748           subroutine daxpy(n,da,dx,incx,dy,incy)
749     c
750     c     constant times a vector plus a vector.
751     c     uses unrolled loops for increments equal to one.
752     c     jack dongarra, linpack, 3/11/78.
753     c
754           double precision dx(*),dy(*),da
755           integer i,incx,incy,ix,iy,m,mp1,n
756     c
757           if(n.le.0)return
758           if (da .eq. 0.0d0) return
759           if(incx.eq.1.and.incy.eq.1)go to 20
760     c
761     c        code for unequal increments or equal increments
762     c          not equal to 1
763     c
764           ix = 1
765           iy = 1
766           if(incx.lt.0)ix = (-n+1)*incx + 1
767           if(incy.lt.0)iy = (-n+1)*incy + 1
768           do 10 i = 1,n
769             dy(iy) = dy(iy) + da*dx(ix)
770             ix = ix + incx
771             iy = iy + incy
772        10 continue
773           return
774     c
775     c        code for both increments equal to 1
776     c
777     c
778     c        clean-up loop
779     c
780        20 m = mod(n,4)
781           if( m .eq. 0 ) go to 40
782           do 30 i = 1,m
783             dy(i) = dy(i) + da*dx(i)
784        30 continue
785           if( n .lt. 4 ) return
786        40 mp1 = m + 1
787           do 50 i = mp1,n,4
788             dy(i) = dy(i) + da*dx(i)
789             dy(i + 1) = dy(i + 1) + da*dx(i + 1)
790             dy(i + 2) = dy(i + 2) + da*dx(i + 2)

901           double precision function dnrm2 ( n, dx, incx)
902           integer          next
903           double precision   dx(*), cutlo, cuthi, hitest, sum, xmax,zero,one
904           data   zero, one /0.0d0, 1.0d0/
905     c
906     c     euclidean norm of the n-vector stored in dx() with storage
907     c     increment incx .
908     c     if    n .le. 0 return with result = 0.
909     c     if n .ge. 1 then incx must be .ge. 1
910     c
911     c           c.l.lawson, 1978 jan 08
912     c
913     c     four phase method     using two built-in constants that are
914     c     hopefully applicable to all machines.
915     c         cutlo = maximum of  dsqrt(u/eps)  over all known machines.
916     c         cuthi = minimum of  dsqrt(v)      over all known machines.
917     c     where
918     c         eps = smallest no. such that eps + 1. .gt. 1.
919     c         u   = smallest positive no.   (underflow limit)
920     c         v   = largest  no.            (overflow  limit)
921     c
922     c     brief outline of algorithm..
923     c
924     c     phase 1    scans zero components.
925     c     move to phase 2 when a component is nonzero and .le. cutlo
926     c     move to phase 3 when a component is .gt. cutlo
927     c     move to phase 4 when a component is .ge. cuthi/m
928     c     where m = n for x() real and m = 2*n for complex.
929     c
930     c     values for cutlo and cuthi..
931     c     from the environmental parameters listed in the imsl converter
932     c     document the limiting values are as follows..
933     c     cutlo, s.p.   u/eps = 2**(-102) for  honeywell.  close seconds are
934     c                   univac and dec at 2**(-103)
935     c                   thus cutlo = 2**(-51) = 4.44089e-16
936     c     cuthi, s.p.   v = 2**127 for univac, honeywell, and dec.
937     c                   thus cuthi = 2**(63.5) = 1.30438e19
938     c     cutlo, d.p.   u/eps = 2**(-67) for honeywell and dec.
939     c                   thus cutlo = 2**(-33.5) = 8.23181d-11
940     c     cuthi, d.p.   same as s.p.  cuthi = 1.30438d19
941     c     data cutlo, cuthi / 8.232d-11,  1.304d19 /
942     c     data cutlo, cuthi / 4.441e-16,  1.304e19 /
943           data cutlo, cuthi / 8.232d-11,  1.304d19 /
944     c
945           if(n .gt. 0) go to 10
946              dnrm2  = zero
947              go to 300
948     c
949        10 assign 30 to next
950           sum = zero
951           nn = n * incx
952     c                                                 begin main loop
953           i = 1
954        20    go to next,(30, 50, 70, 110)
955        30 if( dabs(dx(i)) .gt. cutlo) go to 85
956           assign 50 to next
957           xmax = zero
958     c
959     c                        phase 1.  sum is zero
960     c
961        50 if( dx(i) .eq. zero) go to 200
962           if( dabs(dx(i)) .gt. cutlo) go to 85
963     c
964     c                                prepare for phase 2.
965           assign 70 to next
966           go to 105
967     c
968     c                                prepare for phase 4.
969     c
970       100 i = j
971           assign 110 to next
972           sum = (sum / dx(i)) / dx(i)
973       105 xmax = dabs(dx(i))
974           go to 115
975     c
976     c                   phase 2.  sum is small.
977     c                             scale to avoid destructive underflow.
978     c
979        70 if( dabs(dx(i)) .gt. cutlo ) go to 75
980     c
981     c                     common code for phases 2 and 4.
982     c                     in phase 4 sum is large.  scale to avoid overflow.
983     c
984       110 if( dabs(dx(i)) .le. xmax ) go to 115
985              sum = one + sum * (xmax / dx(i))**2
986              xmax = dabs(dx(i))
987              go to 200
988     c
989       115 sum = sum + (dx(i)/xmax)**2
990           go to 200
991     c
992     c
993     c                  prepare for phase 3.
994     c
995        75 sum = (sum * xmax) * xmax
996     c
997     c
998     c     for real or d.p. set hitest = cuthi/n
999     c     for complex      set hitest = cuthi/(2*n)
1000     c
1001        85 hitest = cuthi/float( n )
1002     c
1003     c                   phase 3.  sum is mid-range.  no scaling.
1004     c
1005           do 95 j =i,nn,incx
1006           if(dabs(dx(j)) .ge. hitest) go to 100
1007        95    sum = sum + dx(j)**2
1008           dnrm2 = dsqrt( sum )
1009           go to 300
1010     c
1011       200 continue
1012           i = i + incx
1013           if ( i .le. nn ) go to 20
1014     c
1015     c              end of main loop.
1016     c
1017     c              compute square root and adjust for scaling.
1018     c
1019           dnrm2 = xmax * dsqrt(sum)
1020       300 continue
1021           return
1022           end
